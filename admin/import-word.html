  if (!token) { setPublishStatus('⚠️ Entre ton token GitHub.', 'warn'); return; }

  var articleObj = buildArticleObject();
  if (!articleObj.title || !articleObj.slug) {
    setPublishStatus('⚠️ Remplis le titre et le slug avant de publier.', 'warn'); return;
  }

  setPublishStatus(' Récupération de blog.json…', '');
  document.getElementById('btn-publish').disabled = true;

      fetch('/article.html?_=' + Date.now())
    ]);
